	\chunk{
		\quess{18}{
			Why must a matrix be square if it has an inverse? \\
			Let \textbf{A} be a matrix of size $p\times q$ and $\textbf{A}^{-1}$ of size $q\times r$. \\
			$\textbf{A}\textbf{A}^{-1}=\textbf{I}=\textbf{A}^{-1}\textbf{A}$ \\
			$\underbrace{\textbf{A}\textbf{A}^{-1}}_{p\times r}=\textbf{A}^{-1}\textbf{A}$ \\
			In order for $\textbf{A}^{-1}\textbf{A}$ to be valid $p=r$. \\
			$\underbrace{\textbf{A}\textbf{A}^{-1}}_{r\times r}=\underbrace{\textbf{A}^{-1}\textbf{A}}_{q\times q}$ \\
			$p=r=q$ therefore \textbf{A} must be square. \\
		}
	}
